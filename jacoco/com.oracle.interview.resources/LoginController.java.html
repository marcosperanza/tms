package com.oracle.interview.resources;

import com.oracle.interview.auth.SimpleDatabaseAuthenticator;
import com.oracle.interview.db.UserRepository;
import com.oracle.interview.db.entity.User;
import io.dropwizard.auth.Auth;
import io.dropwizard.hibernate.UnitOfWork;
import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.Parameter;
import io.swagger.v3.oas.annotations.media.Content;
import io.swagger.v3.oas.annotations.media.Schema;
import io.swagger.v3.oas.annotations.responses.ApiResponse;
import io.swagger.v3.oas.annotations.responses.ApiResponses;
import io.swagger.v3.oas.annotations.tags.Tag;

import javax.ws.rs.GET;
import javax.ws.rs.PUT;
import javax.ws.rs.Path;
import javax.ws.rs.core.Cookie;
import javax.ws.rs.core.NewCookie;
import javax.ws.rs.core.Response;
import java.nio.charset.StandardCharsets;
import java.util.Base64;
import java.util.Optional;


/**
 * Quick and dirty PoC for basic auth.
 * is based on Basic auth
 */
@Path("/auth")
@Tag(name = "LoginController")
public class LoginController {

    UserRepository daoUser;

    public LoginController(UserRepository daoUser) {
        this.daoUser = daoUser;
    }


    /**
     *  Execute a simple login based on Cookie. It sets a SSID
     */
    @Operation(summary = "Simple Login, tht uses basic auth", tags = "LoginController")
    @ApiResponses(value = {
            @ApiResponse(responseCode = "200", description = "Success|OK"),
            @ApiResponse(responseCode = "401", description = "UNAUTHORIZED"),
            @ApiResponse(responseCode = "500", description = "In case of error")
    })
    @PUT
    @Path("/login")
    @UnitOfWork
    public Response simpleBasicLogin(User u) {
        Optional<User> user = daoUser.getUser(u.getUsername());
        if (user.isPresent() &&  u.getPassword().equals(user.get().getPassword())) {
            String encodeToString = Base64.getEncoder().encodeToString((u.getUsername() + ":" + u.getPassword()).getBytes(StandardCharsets.UTF_8));
            Cookie c = new Cookie("SSID", encodeToString, "/", null);
            NewCookie cookie = new NewCookie(c);
            return Response.ok().entity(u.getUsername()).cookie(cookie).build();
        }

        return Response.status(Response.Status.UNAUTHORIZED).build();
    }

    @Operation(summary = "Get logged username", tags = "LoginController")
    @ApiResponses(value = {
            @ApiResponse(responseCode = "200", description = "Success|OK", content = @Content(schema = @Schema(implementation = String.class))),
            @ApiResponse(responseCode = "401", description = "UNAUTHORIZED"),
            @ApiResponse(responseCode = "500", description = "In case of error")
    })
    @GET
    @Path("/username")
    public Response loggedUserName(@Parameter(name = "user", hidden = true) @Auth Optional<User> user) {
        return Response.ok().entity(user.orElse(SimpleDatabaseAuthenticator.GUEST_USER).getUsername()).build();
    }


    @Operation(summary = "Logout", tags = "LoginController")
    @ApiResponses(value = {
            @ApiResponse(responseCode = "200", description = "Success|OK", content = @Content(schema = @Schema(implementation = String.class))),
            @ApiResponse(responseCode = "401", description = "UNAUTHORIZED"),
    })
    @GET
    @Path("/logout")
    public Response logout() {
        Cookie c = new Cookie("SSID", "", "/", null);
        NewCookie cookie = new NewCookie(c, "", 0, true);
        return Response.status(Response.Status.OK).cookie(cookie).build();
    }
}
